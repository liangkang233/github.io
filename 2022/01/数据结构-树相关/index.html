TreeNode *OrderToTree(int L_Pre, int R_Pre, int L_In, int R_In, vector<int> &Pre, vector<int> &In) 
{   
    if (L_Pre > R_Pre || L_In > R_In)
        return NULL;
    TreeNode *root = new TreeNode(Pre[L_Pre]);   //å‰åºæ’åˆ—ä¸­ æœ€å·¦ä½ä¸ºæ ¹èŠ‚ç‚¹

    return root;
}
