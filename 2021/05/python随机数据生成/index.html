class Solution {

    int search(vector<int>& nums, int target) {
        int n = (int)nums.size();
        if (!n) 
            return -1;
        else if (n == 1) 
            return nums[0] == target ? 0 : -1;

        int l = 0, r = n - 1;
        while (l <= r) {
            int mid = (l + r) / 2;
            if (nums[mid] == target) return mid;    // targetåˆ¤æ–­æ˜¯å¦ä¸ºmid
            if (nums[0] <= nums[mid]) {             // è¯´æ˜0åˆ°midæ˜¯å‡åº

                if (nums[mid] < target && target <= nums[n - 1])  // targetåœ¨è¯¥å‡åºä¸­
                    l = mid + 1;
                else 
                    r = mid - 1;

        }
        return -1;
    }
